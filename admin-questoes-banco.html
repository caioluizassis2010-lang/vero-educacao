📄 Modelo de importação JSON
Cada questão deve seguir este formato. Campos obrigatórios: disciplina, enunciado, gabarito.
[ { "sistema": "concursos", // concursos | medicina | enem "banca": "CESPE", // nome da banca/vestibular "ano": 2023, // ano da prova "cargo": "Analista Judiciário", // cargo (concursos) "orgao": "STJ", // órgão (concursos) "vestibular": "FUVEST", // vestibular (medicina) "disciplina": "Direito Constitucional", "assunto": "Direitos Fundamentais", "subassunto": "Remédios Constitucionais", "dificuldade": 3, // 1(fácil) a 5(muito difícil) "tipo": "multipla_escolha", // multipla_escolha | certo_errado "texto_base": "Texto de apoio...", // opcional "enunciado": "Texto da questão...", "opcoes": [ "Alternativa A", "Alternativa B", "Alternativa C", "Alternativa D", "Alternativa E" ], "gabarito": "C", // letra correta ou C/E "explicacao": "Explicação...", // opcional mas recomendado "tags": ["constitucional", "direitos"] } ]
